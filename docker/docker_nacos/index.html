Docker_nacos

December 19, 2023

Docker启动nacos#

下载镜像#

# 默认下载latest
docker pull nacos/nacos-server

启动镜像#

启动一个模板镜像#

启动模板镜像主要用于获取配置文件。

#nacos2.0以后的版本需要多开放9848和9849两个端口
docker run -itd --env MODE=standalone --name nacos -d -p 8848:8848 -p 9848:9848 -p 9849:9849 nacos/nacos-server

创建本地目录#

mkdir -p /home/slg/nacos/logs && cd /home/slg/nacos

copy模板容器的配置文件#

#将naco模板容器中的配置copy到本地
docker cp nacos:/home/nacos/conf ./

删除模板容器#

docker rm -f nacos

添加授权#

开启授权

#在配置文件application.properties添加如下语句开启授权
#默认的nacos用户名密码nacos，nacos
# 启用鉴权
nacos.core.auth.enabled=true
# 设置登录用户名和密码（假设为 admin/admin）
nacos.core.auth.username=admin
#这是 admin 的 BCrypt 哈希值。实际应用中，您可以使用类似的方式生成并指定密码的哈希值。
#一般状态下通过使用默认用户名密码进入，然后修改nacos账户的密码就可以。
nacos.core.auth.password=$2a$10$gXDu6T1M1zCZK.kPaqOrPucd7D76.y44/Z5k2qqfLPt96iTsTx3ZK

设置账户密码

#修改如下语句
#通过token访问nacos配置如下
#自定义密钥时，推荐将配置项设置为Base64编码的字符串，且原始密钥长度不得低于32字符
nacos.core.auth.plugin.nacos.token.secret.key=${NACOS_AUTH_TOKEN:SecretKey012345678901234567890123456789012345678901234567890123456789}
#校验的key
nacos.core.auth.server.identity.key=${NACOS_AUTH_IDENTITY_KEY:example_key}
#校验的value
nacos.core.auth.server.identity.value=${NACOS_AUTH_IDENTITY_VALUE:example_value}

identity配置的作用

#当直接访问nacos接口
http://ip:8846/nacos/v1/cs/configs?dataId=&group=&appName=&config_tags=&pageNo=1&pageSize=10&tenant=&search=blur&username=nacos

设置数据信息

#这里必须开启持久化为mysql数据库
spring.sql.init.platform=${SPRING_DATASOURCE_PLATFORM:mysql}
db.num=${MYSQL_DATABASE_NUM:1}
db.url.0=jdbc:mysql://${MYSQL_SERVICE_HOST:xxx}:${MYSQL_SERVICE_PORT:3306}/${MYSQL_SERVICE_DB_NAME:nacos_conf}?${MYSQL_SERVICE_DB_PARAM:characterEncoding=utf8&connectTimeout=1000&socketTimeout=3000&autoReconnect=true&useSSL=false}
db.user.0=${MYSQL_SERVICE_USER:root}
db.password.0=${MYSQL_SERVICE_PASSWORD:xxx}

正式启动容器#

docker run -p 8848:8848 \
-p 9848:9848 -p 9849:9849 \
-e NACOS_AUTH_ENABLE=true \
-e MODE=standalone \
-e JVM_XMS=1024m \
-e JVM_XMX=1024m \
-e JVM_XMN=1024m \
-v /home/slg/nacos/logs:/home/nacos/logs \
-v /home/slg/nacos/conf/:/home/nacos/conf \
--privileged=true \
--restart=always \
--name nacos \
-itd nacos/nacos-server
