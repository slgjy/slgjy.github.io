2.实现Store#

实现Store来自定义验证码存储。

type Store interface {
	// Set sets the digits for the captcha id.
	Set(id string, value string)

	// Get returns stored digits for the captcha id. Clear indicates
	// whether the captcha must be deleted from the store.
	Get(id string, clear bool) string
	
    //Verify captcha's answer directly
	Verify(id, answer string, clear bool) bool
}

实现store存储

package captcha

import (
	"context"
	"fmt"
	"log"
	"statistical_data/global"
	"time"
)

var ctx = context.Background()

const CAPTCHA = "captcha:"

type RedisStore struct {
}

// 实现Store接口来自定义验证码存储

// Set set a capt
func (r RedisStore) Set(id string, value string) error {
	key := CAPTCHA + id
	err := global.Redis.Set(key, value, time.Minute*2).Err()
	if err != nil {
		log.Println(err.Error())
		panic(err)
	}
	return err
}

// Get get a capt
func (r RedisStore) Get(id string, clear bool) string {
	key := CAPTCHA + id
	val, err := global.Redis.Get(key).Result()
	if err != nil {
		fmt.Println(err)
		return ""
	}
	if clear {
		err := global.Redis.Del(key).Err()
		if err != nil {
			fmt.Println(err)
			return ""
		}
	}
	return val
}

// Verify verify a capt
func (r RedisStore) Verify(id, answer string, clear bool) bool {
	v := RedisStore{}.Get(id, clear)
	return v == answer
}

Data URI scheme#

生成验证码的字符串前缀data:image/png;base64，放到前端页面的图片的src属性中。

Data URI scheme是什么#

Data URI scheme 是在 RFC2397 中定义的，目的是将一些小的数据，直接嵌入到网页中，从而不用再从外部文件载入。减少对 HTTP 的请求次数。达到优化网页的效果，与之相对的通过http请求直接获取的成为 HTTP URI scheme

Data URI scheme 的语法#

  • data 表示取得数据的协定名称；

  • image/png 是数据类型名称；

  • base64 是数据的编码方法，逗号后面就是这个image/png文件base64编码后的数据
