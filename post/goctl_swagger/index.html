Goctl-swagger#

goctl-swagger是一个生成文档的插件。

官网地址

使用步骤#

前置条件具备

  1. goland环境
  2. goctl环境
  3. docker环境(用于启动swaggerui)

1.下载插件#

go install github.com/zeromicro/goctl-swagger@latest

2.生成api文件#

#UserApi生成的swagger文件名
#-host后端请求地址和端口
#-api 编写.api文件路径
#-dir 输出目录默认. 当前目录下
goctl api plugin -plugin goctl-swagger="swagger -filename UserApi.json -host 127.0.0.1:8080" -api user.api -dir .

3.docker启动swaggerui#

在linux环境下进行

将生成的swagger接口文件上传到linux机器上

1.下载镜像#

docker pull swaggerapi/swagger-ui

2.启动并配置生成的api文件#

#-e SWAGGER_JSON=/api/userApi.json  配置swaggerUI的接口文档路径(自定义路径，只需要将实际的api文件路径挂载到该配置上)
#-v /home/slg/swagger/userApi.json:/api/userApi.json  将实际的api接口文件挂载到配置的路径下即可
sudo docker run -itd --name swaggerUi \
-p 8083:8080 \
-e SWAGGER_JSON=/api/userApi.json \
-v /home/slg/swagger/userApi.json:/api/userApi.json \
swaggerapi/swagger-ui

3.访问#

自己的机器ip加上容器映射的端口http://ip:port

跨域处理#

swaggerUi测试出现跨域问题#

后端代码在Windows机器上运行，swaggerui容器运行在Linux虚拟机上进行接口测试时，出现了跨域报错

添加跨域处理中间件#

一般跨域中间件都编写在common公共基础包内

package middleware

import "net/http"

// CorsMiddleware 跨域请求处理中间件
type CorsMiddleware struct {
}

// NewCorsMiddleware 新建跨域请求处理中间件
func NewCorsMiddleware() *CorsMiddleware {
	return &CorsMiddleware{}
}

// Handle 跨域请求处理
func (m *CorsMiddleware) Handle(next http.HandlerFunc) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		setHeader(w)

		// 放行所有 OPTIONS 方法
		if r.Method == "OPTIONS" {
			w.WriteHeader(http.StatusNoContent)
			return
		}

		// 处理请求
		next(w, r)
	}
}

// Handler 跨域请求处理器
func (m *CorsMiddleware) Handler() http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		setHeader(w)

		if r.Method == "OPTIONS" {
			w.WriteHeader(http.StatusNoContent)
		} else {
			w.WriteHeader(http.StatusNotFound)
		}
	})
}

// setHeader 设置响应头
func setHeader(w http.ResponseWriter) {
	w.Header().Set("Access-Control-Allow-Origin", "*")
	w.Header().Set("Access-Control-Allow-Headers", "*")
	w.Header().Set("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE, OPTIONS, PATCH")
	w.Header().Set("Access-Control-Expose-Headers", "Content-Length, Content-Type, Access-Control-Allow-Origin, Access-Control-Allow-Headers")
	w.Header().Set("Access-Control-Allow-Credentials", "true")
}

应用中间件#

func main() {
	flag.Parse()
	var c config.Config
	conf.MustLoad(*configFile, &c)
    //创建服务器添加中间件
	server := rest.MustNewServer(c.RestConf, rest.WithNotAllowedHandler(middleware.NewCorsMiddleware().Handler()))
	defer server.Stop()
    // 使用中间件
	server.Use(middleware.NewCorsMiddleware().Handle)
	ctx := svc.NewServiceContext(c)
	handler.RegisterHandlers(server, ctx)
	fmt.Printf("Starting server at %s:%d...\n", c.Host, c.Port)
	server.Start()
}
