go单元测试#

go test工具#

go test命令是一个执行测试代码的go程序，在包目录内，所有以*_test.go为后缀的源码文件，不会被go build编译到最终的可执行文件。

func checkName(name string) {
	if name != "admin" {
		panic(errors.New("用户名不正确！！"))
	}
}

package main

import "testing"

// 函数测试必须是Test开头接收*testing.T参数
func TestCheckName(t *testing.T) {
	nameStr := "userName"
	// 被测试的函数可以是小写开头
	checkName(nameStr)

go test常用命令#

  1. 执行单个*_test.go文件下的所有方法

    go test -v demo_test.go demo.go

  2. 执行单个*_test.go文件下的指定方法

    go test -v -test.run TestDemoMethod

  3. 测试覆盖率

    go test -cover
// 将数据输出到指定文件
go test -cover -coverprofile=c.out
// 执行如下命令打开生成的文件
go tool cover -html=c.out

  4. 基准测试

    //demoMtehod被测试的函数
go test -bench=demoMtehod
