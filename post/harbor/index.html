配置内核参数#

# cat > /etc/sysctl.conf << EOF
net.ipv4.ip_forward = 1
net.bridge.bridge-nf-call-ip6tables = 1
net.bridge.bridge-nf-call-iptables = 1
EOF
sysctl -p

net.ipv4.ip_forward = 1: 开启路由转发，不配置该参数，主机重启后，服务状态正常，却无法访问到服务器。

使用cfssl工具配置Harbor证书#

默认的Harbor不附带证书，生产环境为了安全需要使用HTTPS,如果启用Content Trust with Notary 以正确签署所有图像，则必须使用 HTTPS。要配置 HTTPS，必须创建 SSL 证书。您可以使用受信任的第三方 CA 签署的证书，也可以使用自签名证书。本节介绍如何使用cfssl创建 CA，以及如何使用您的 CA 签署服务器证书和客户端证书。官方使用的是openssl工具来生成证书。

安装cfssl工具#

安装步骤已经记录在其他文档

生成默认的配置#

# 通过cfssl命令生成的默认配置写入ca-config.json文件
cfssl print-defaults  config > ca-config.json

修改生成的ca-config.json文件

{
    "signing": {
        "default": {
            "expiry": "87600h"
        },
        "profiles": {
            "harbor": {
                "expiry": "87600h",
                "usages": [
                    "signing",
                    "key encipherment",
                    "server auth",
                    "client auth"
                ]
            }
        }
    }
}

生成csr请求文件#

cfssl  print-defaults csr  > ca-csr.json

修改生成的ca-csr.json文件

{
    "CN": "harbor",
    "hosts": [
        "10.62.131.1"
    ],
    "key": {
        "algo": "rsa",
        "size": 2048
    },
    "names": [
        {
            "C": "CN",
            "ST": "Beijing",
            "L": "Beijing"
        }
    ]
}

hosts：包含授权范围，不在此范围的节点或者服务使用此证书会报证书不匹配错误，

key： 指代加密算法，一般使用ras非对称加密算法

初始化CA#

cfssl  gencert  -initca  ca-csr.json  |  cfssljson  -bare   ca

生成服务器证书#

创建并修改服务器证书请求文件#
cfssl  print-defaults csr >  harbor-csr.json

使用请求文件根据CA配置颁发证书#

切换到证书所在目录执行

cfssl gencert -ca=ca.pem -ca-key=ca-key.pem \
-config=ca-config.json \
-profile=harbor  harbor-csr.json | cfssljson -bare  harbor

-config 指定 CA证书机构的配置文件

-profile 指定使用CA配置文件中那个模块

harbor配置文件指定使用证书#

vim harbor.yml

https:
  port: 443
  certificate: /etc/docker/CA/harbor.pem
  private_key: /etc/docker/CA/harbor-key.pem

部署harbor#

下载地址

下载并解压#

harbor.v2.5.6.tar.gz 镜像包

install.sh会自动导入该镜像，这里也就是为什么需要docker了

使用模板文件创建配置文件#
[root@IT4ITothers005 harbor]# cp harbor.yml.tmpl harbor.yml.tmpl.bak
[root@IT4ITothers005 harbor]# mv harbor.yml.tmpl harbor.yml
#修改配置文件
vim harbor.yml

hostname: 10.62.131.1
http:
  port: 8082
https:
  port: 443
#这里配置的就是cfssl生成的秘钥证书信息
  certificate: /home/cfssl/harbor.pem
  private_key: /home/cfssl/harbor-key.pem
# 初始的harbor密码不能修改
harbor_admin_password: Harbor12345
database:
  #数据库密码(暂时不知道何用,没有看见数据库链接信息)
  password: xxxxx
  max_idle_conns: 100
  max_open_conns: 900
# 数据挂载地址
data_volume: /data
运行脚本prepare脚本启用https#
# harbor解压目录执行
./prepare
运行install.sh安装harbor#
# 将镜像加载到docker引擎
./install.sh

验证服务是否正常#
docker-compose ps

Harbor启停#

#关闭
docker-compose down -v
#启动,需要在harbor解压目录下执行，该目录下有docker-compose.yml文件
docker-compose up -d
