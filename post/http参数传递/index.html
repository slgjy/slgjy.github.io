queryParam参数传递#

func main() {
	uri := "http://localhost:8080/test"
	request, err := http.NewRequest(http.MethodGet, uri, nil)
	if err != nil {
		fmt.Println("构造请求报错!!")
	}
	// 设置header参数
	request.Header.Add("Cookie", "CookieStr")
	// 设置路径中的查询参数
	queryParam := url.Values{}
	queryParam.Add("name", "张三")
	queryParam.Add("age", "12")
	// 将参数编码后添加到请求中
	request.URL.RawQuery = queryParam.Encode()
	client := http.Client{}
	resp, err := client.Do(request)
	if err != nil {
		fmt.Println("请求接口报错!!")
	}
	defer resp.Body.Close()
}

type user struct {
	// json解析需要大写
	Name string
	Age  int8
	Sex  string
}

func main() {
	uri := "http://localhost:8080/test"
	user1 := user{"张三", 24, "man"}
	marshal, _ := json.Marshal(user1)
	request, err := http.NewRequest(http.MethodPost, uri, bytes.NewBuffer(marshal))

form表单参数传递#

func main(){
	filePath := "/text.txt"
	uri := "http://localhost:8080/test"
	open, _ := os.Open(filePath)
	payload := &bytes.Buffer{}
	writer := multipart.NewWriter(payload)
	// 表单参数信息
	file, err := writer.CreateFormFile("file", filepath.Base(filePath))
	if err != nil {
		panic(errors.New("创建form表单失败！！"))
	}
	_, _ = io.Copy(file, open)
	// 这个是必须的，添加消息结尾的boundary
	writer.Close()
	request, err := http.NewRequest(http.MethodPost, uri, payload)
	if err != nil {
		panic(err)
	}
	request.Header.Set("Content-Type", writer.FormDataContentType())
	client := http.Client{}
	resp, err := client.Do(request)
	defer resp.Body.Close()
}
