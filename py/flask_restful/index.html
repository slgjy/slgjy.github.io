flask _restful是flask的一个扩展，添加了快速扩展REST API的支持

安装#

pip install flask-restful

快速入门#

目录结构

user.py

from flask import Blueprint
from flask_restful import Resource, Api

bp = Blueprint('user', __name__, url_prefix='/user')
api = Api(bp)

class UserView(Resource):
    def get(self):
        return {"username": "slg"}

    def post(self):
        return {"username": "slg_post"}

api.add_resource(UserView, '/list')

from flask import Blueprint
from flask_restx import Resource, Api, marshal, fields, marshal_with

from models import db, User

# 1.创建蓝图对象
user_bp = Blueprint('user_bp', __name__, url_prefix='/user')

api = Api(user_bp)

# 序列化规则
fields = {
    'username': fields.String(attribute='name'),  # 指定对应的模型属性，将响应字段重命名
    'id': fields.Integer(default=10),  # 设置默认值
}
class DemoViewResource(Resource):
    # 使用装饰器处理
    @marshal_with(fields, envelope='msgData')
    def post(self):
        users = db.session.query(User).all()
        print('post')
        return users
api.add_resource(DemoViewResource, '/demo')

==自定义响应==#

通过api.representation()装饰器自定义返回数据

实现方式一#

直接在视图写入

@api.representation('application/json')
def output_json(data: object, code: object, headers: object = None) -> object:
    settings = current_app.config.get('RESTFUL_JSON', {})

    if current_app.debug:
        settings.setdefault('indent', 4)
        settings.setdefault('sort_keys', not PY3)

    # 添加json外层包装
    if 'message' not in data:  # 判断是否设置了自定义的错误信息
        data = {
            'code': 200,
            'message': 'ok',
            'data': data
        }

    dumped = dumps(data, **settings) + "\n"

    resp = make_response(dumped, code)
    resp.headers.extend(headers or {})
    return resp

实现output_json方法

from json import dumps

from flask import current_app, make_response
from six import PY3


def output_json(data: object, code: object, headers: object = None) -> object:
    settings = current_app.config.get('RESTFUL_JSON', {})

    if current_app.debug:
        settings.setdefault('indent', 4)
        settings.setdefault('sort_keys', not PY3)

        data = {
            'code': 200,
            'message': 'ok',
            'data': data
        }

    dumped = dumps(data, **settings) + "\n"
