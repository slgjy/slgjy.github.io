class Config:
    DEBUG = True
    """
    使用mysqldb驱动协议名mysql
    使用pymysql驱动协议名为mysql+pymysql
    """
    # 协议名://用户名:密码@主机名:端口/数据库名
    SQLALCHEMY_DATABASE_URI = 'mysql+pymysql://username:password@x.x.x.x:3306/demo'
    SQLALCHEMY_TRACK_MODIFICATIONS = False  # 追踪数据的修改信号
    SQLALCHEMY_ECHO = True  # 是否在控制台打印输出sql语句
    JSON_AS_ASCII = False  # 不转移中文字符

初始化方式

#第一种
from flask import Flask
from flask_sqlalchemy import SQLAlchemy
app = Flask(__name__)
#直接关联flask应用
db = SQLAlchemy(app)
#第二种
from flask import Flask
from flask_sqlalchemy import SQLAlchemy


# 方式2: 初始化组件对象, 延后关联Flask应用
db = SQLAlchemy()
def create_app(config_type):
    # 创建应用
    flask_app = Flask(__name__)
    # 加载配置
    config_class = config_dict[config_type]
    flask_app.config.from_object(config_class)
    # 关联flask应用
    db.init_app(app)
    return flask_app

构建模型类#

#模型必须集成db.Model
class User(db.Model):
    __tablename__ = "user"
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(50))
    address = db.relationship('Address', backref='user', uselist=False)

# 序列化规则
fields = {
    'username': fields.String(attribute='name'),  # 指定对应的模型属性
    'id': fields.Integer(default=10),  # 设置默认值
}

# 通过导入自定义的函数实现
api.representation('application/json')(output_json)

class DemoViewResource(Resource):
    def get(self):
        # 创建解析器对象
        parser = reqparse.RequestParser()
        # 利用解析器添加需要验证的参数
        parser.add_argument('name', type=str, help='not find name param')
        args = parser.parse_args()
        # users = db.session.query(User).filter(User.name == args.get('name')).all()
        # 原生SQL查询
        users = db.session.query(User).from_statement(text("SELECT * FROM user where name=:name")).params(
            name=args.get('name')).all()
        # SQLAlchemy使用sql语句查询,需要test组装数据，且参数传递需要注意
        sql = text("SELECT * FROM user where name=:name").params(name=args.get('name'))
        result = db.session.execute(sql)
        users = result.fetchall()
        # marshal函数可以按照指定的序列化规则将 模型对象 转为 字典
        return marshal(users, fields)
api.add_resource(DemoViewResource, '/demo')
